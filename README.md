```javascript

  destination: Cesium.Rectangle.fromDegrees(
    westLng - (eastLng - westLng) / 2,
    southLat - (northLat - southLat) / 2,
    eastLng + (eastLng - westLng) / 2,
    northLat + (northLat - southLat) / 2
  ),
});
```
